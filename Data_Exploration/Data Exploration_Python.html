    x4.append(round(i,2))
    i=i+0.1

y2=[0] * len(x4)

    y1[i] = round(2 + 3 * x4[i] - (0.25 * math.pow(x4[i],2)),4)
    y2[i] = round(3 + 3.3 * x4[i] - 0.3 * math.pow(x4[i],2), 4)
    i += 1
    
print(y1)
print('\n')
print(y2)


plt.plot(x4, y1, 'b')
plt.plot(x4, y2, 'r')
plt.show()
